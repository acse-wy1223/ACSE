163 * allowing for finer mutations as the number of generations increases.
164 *
165 * @param individual A reference to the individual (vector of ints) to mutate.
166 * @param mutation_rate The mutation rate to apply.
167 * @param max_value The maximum value for any gene in the individual.
168 * @param currentGeneration The current generation number in the genetic algorithm.
169 * @param maxGenerations The maximum number of generations expected to run.
170 */
171void NonUniform_Mutation(std::vector<int>& individual, double mutation_rate, int max_value, int currentGeneration, int maxGenerations) {
172
173 for (int& gene : individual) {
174 if (generate_random_number(0.0, 1.0) < mutation_rate) {
175 double delta = (generate_random_number(0.0, 1.0) < 0.5) ? gene : max_value - gene;
176 double b = 5;
177 double r = generate_random_number(0.0, 1.0);
178
179 double change = delta * (1 - pow(r, pow((1 - double(currentGeneration) / maxGenerations), b)));
180 gene = (generate_random_number(0.0, 1.0) < 0.5) ? gene - static_cast<int>(change) : gene + static_cast<int>(change);
181
182 if (gene < 0) gene = 0;
183 if (gene > max_value) gene = max_value;
184 }
185 }
