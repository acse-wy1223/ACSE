145/**
146 * @brief Calculates the residence time of materials in a unit.
147 *
148 * @param constants The constants used for calculation.
149 * @param Fg Flow rate of gerardium.
150 * @param Fw Flow rate of waste.
151 * @return The calculated residence time.
152 */
153double calculate_residence_time(const Calculate_constants &constants, double Fg, double Fw)
154{
155 double total_mass_flow_rate = Fw + Fg;
156 total_mass_flow_rate = std::max(total_mass_flow_rate, 1e-10);
157 double volume_flow_rate = total_mass_flow_rate / constants.rho;
158 double tau = constants.phi * constants.V / volume_flow_rate;
159 return tau;
160};

162/**
163 * @brief Calculates the recovery of materials in a unit.
164 *
165 * @param constants The constants used for calculation.
166 * @param tau The residence time of materials in the unit.
167 * @return The calculated recovery.
168 */
169struct Recovery calculate_recovery(const Calculate_constants &constants, double tau)
170{
171 double rcg = (constants.k_concentrate_gerardium * tau) / (1 + (constants.k_concentrate_gerardium + constants.k_inter_gerardium) * tau);
172 double rcw = (constants.k_concentrate_waste * tau) / (1 + (constants.k_concentrate_waste + constants.k_inter_waste) * tau);
173 double rig = (constants.k_inter_gerardium * tau) / (1 + (constants.k_concentrate_gerardium + constants.k_inter_gerardium) * tau);
174 double riw = (constants.k_inter_waste * tau) / (1 + (constants.k_concentrate_waste + constants.k_inter_waste) * tau);
175
176 struct Recovery recoveries = {rcg, rcw, rig, riw};
177 return recoveries;
178};

180/**
181 * @brief Calculates the flow rates of materials in a unit.
182 *
183 * @param constants The constants used for calculation.
184 * @param recovery The recovery of materials in the unit.
185 * @param init_Fg Initial flow rate of gerardium.
186 * @param init_Fw Initial flow rate of waste.
187 * @return The calculated flow rates.
188 */
189std::vector<double> calculate_flow_rate(const Calculate_constants &constants, Recovery &recovery, double init_Fg, double init_Fw)
190{
191 double cg = init_Fg * recovery.concentrate_gerardium;
192 double cw = init_Fw * recovery.concentrate_waste;
193 double ig = init_Fg * recovery.inter_gerardium;
194 double iw = init_Fw * recovery.inter_waste;
195 double tg = init_Fg - cg - ig;
196 double tw = init_Fw - cw - iw;
197
198 return {cg, cw, ig, iw, tg, tw};
199};
